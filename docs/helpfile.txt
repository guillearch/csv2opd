NAME
    csv2opd

DESCRIPTION
    # CSV2OPD parses each row of a CSV file into a separate OPD file
    # and names each new file after the value contained in the Name field.
    #
    # The CSV file must have a header and mustn't have an "OPDObject type"
    # colon.
